## 간단한 계산기 만들기

지금까지 배운 내용을 종합해서 간단한 계산기를 만들어볼까요?

python
def calculator(num1, num2, operation):
    if operation == "+":
        return num1 + num2
    elif operation == "-":
        return num1 - num2
    elif operation == "*":
        return num1 * num2
    elif operation == "/":
        if num2 != 0:
            return num1 / num2
        else:
            return "0으로 나눌 수 없습니다!"
    else:
        return "잘못된 연산자입니다."

# 계산기 사용
print(calculator(10, 5, "+"))  # 15
print(calculator(10, 5, "-"))  # 5
print(calculator(10, 5, "*"))  # 50
print(calculator(10, 5, "/"))  # 2.0
